Theorem Name:Unit Package:Logic{
\let x \in \setnatural;
\state \forall x \follows (x < 0) \eq \false;
}
Demonstration { 
    (x = 0) \then (x = 0) \or (x > 0) \eq (x >= 0);

    \false \eq
    \not \true \eq
    \not \exists x \suchthat (x >= 0) \eq
    \not \exists x \suchthat (\not (x < 0)) \eq
    \forall x \follows \not \not (x < 0) \eq
    \forall x \follows (x < 0);
}
