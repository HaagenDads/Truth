Theorem Name:Unit Package:Logic{
\let x \in \setnatural;
\let p \in \boolean;
\let q \in \boolean;
\state \not \forall x \follows (p \implies q) \eq \exists x \suchthat (p \and \not q);
}
Demonstration { 
    \not \forall x \follows (p \implies q) \eq
    \exists x \suchthat \not (p \implies q) \eq
    \exists x \suchthat \not (\not p \or q) \eq
    \exists x \suchthat (\not \not p \and \not q) \eq
    \exists x \suchthat (p \and \not q);


}
