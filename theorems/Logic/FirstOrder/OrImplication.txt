Theorem Name:OrImplication Package:Logic{
\let a \in \boolean;
\let b \in \boolean;
\let r \in \boolean;
\state (a \implies r) \and (b \implies r) = (a \or b) \implies r;
}
Demonstration { 
    (a \or b) \implies r = (\not (a \or b)) \or r = ((\not a) \and (\not b)) \or r
    = ((\not a) \or r) \and ((\not b) \or r) = (a \implies r) \and ((\not b) \or r)
    = (a \implies r) \and (b \implies r);
}
